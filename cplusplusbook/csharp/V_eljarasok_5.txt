using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace V_eljarasok_5
{
    class Program
    {
        static void kep(int n)
        {
            int i;
            Console.Write("  ");
            for (i = 1; i <= n; i++)
                Console.Write(i / 10);
            Console.WriteLine();
            Console.Write("  ");
            for (i = 1; i <= n; i++)
                Console.Write(i % 10);
            Console.WriteLine();
            for (i = 1; i <= n; i++)
                Console.WriteLine("{0}{1}", i / 10, i % 10);
        }
        static void Main(string[] args)
        {
            kep(22);
            Console.ReadLine();
        }
    }
}
