using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace V_eljarasok_3
{
    class Program
    {
        static void elvalaszto(string feladat, bool fel)
        {
            int n = 79;
            if (fel) n /= 2;
            for (int i = 1; i <= n; i++)
                Console.Write('-');
            Console.WriteLine("\n{0}. feladat\n", feladat);
        }
        static void Main(string[] args)
        {
            elvalaszto("3", true);
            elvalaszto("b", false);
            elvalaszto("3.b", true);
            Console.ReadLine();
        }
    }
}
